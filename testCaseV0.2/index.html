DK JS Framework - garnet

Download

http://ssw3131.github.io/garnet/v0.2/garnet.js

Installation

Initialize

돔 로딩 완료 후 '// code' 부분이 실행됩니다. 
            dk( function(){
                // code
            } )

Simple Example

간단 예제 입니다. div 엘리먼트 생성 후 css 속성 입력과 body에 추가하는 코드입니다. 
            dk( function(){
                dk.Dom().S( 'width', 100, 'height', 100, 'bgColor', '#000', '<', 'body' );
            } )

Key, Value

garnet은 기본적으로 key, value 시스템으로 운영됩니다. 다중 파라미터 전달 시 ( key, value ) 쌍으로 연결이 되며 key만 입력 시 value 값이 리턴됩니다. 그 외에는 Dom 객체가 리턴됩니다. 
            // dk.Dom().S( $k, $v );

            var dom = dk.Dom().S( 'width', 100 );
            dom.S( 'height', 100 );
            dom.S( 'position', 'absolute' );

            dom.S( 'left', 100, 'top', 100, 'display', 'none' );

            var domWidth = dk.Dom().S( 'width' );

dk.Dom

엘리먼트 생성 - dk.Dom 객체

기본으로 div 엘리먼트가 생성됩니다. 
            dk.Dom();
파라미터로 문자열 html 코드 전달 시 해당 태크 객체가 생성됩니다. 
            dk.Dom( '
' );
파라미터로 < 뒤 엘리먼트 이름 전달 시 해당 엘리먼트가 생성됩니다.

셀렉터 - css selector

css 셀렉터 문법 그대로 엘리먼트 검색 후 dk.Dom 객체로 리턴됩니다. 
            dk.Dom( '#test' ); // id 검색
            dk.Dom( '.test' ); // 클래스 검색
            dk.Dom( '#test h1' );

엘리먼트 - dom element

엘리먼트 객체 전달 시 해당 객체로 생성된 dk.Dom 객체를 리턴합니다. 
            dk.Dom( document.createElement( 'div' ) );

캐싱 - cache

한번 탐색 된 객체는 돔트리를 재탐색하지 않고 내부 메모리에 캐싱하여 리턴합니다. 
            dk.Dom( '#test' ); // id test 돔트리 탐색
            dk.Dom( '#test' ); // #test 캐싱 리턴

S method

key, value 시스템

key, value 시스템이 적용됩니다.
수치는 px 없이 숫자값으로 입력 가능합니다. 퍼센트는 '10%', 문자열로 입력하면 됩니다. 리턴값은 숫자값과 px 값들은 숫자로 그외는 문자열로 리턴됩니다. 
            // dk.Dom().S( $k, $v );
            var dom = dk.Dom().S( 'left', 100, 'width', 100, 'height', 100 );
            dom.S( 'top', '100%' );
            var left = dom.S( 'left' ); // 100
            var top = dom.S( 'top' ); // '100%'

css vendor prefix

vendor prefix는 자동으로 입력됩니다. 
            // box-shadow, -ms-box-shadow, -moz-box-shadow, -o-box-shadow, -webkit-box-shadow
            dk.Dom().S( 'boxShadow', '0px 0px 10px #62615f inset' );

특수함수

편의성 css 입력 함수를 지원합니다. 
            dk.Dom().S( 'bgColor', $v ) // backgoundColor 대체, rgba 입력 시 ie8 이하 rgb로 대체
            dk.Dom().S( 'bgImg', $v ) // backgroundImage 대체, url 값만 입력
            dk.Dom().S( 'float', $v ) // cssFloat, styleFloat 자동 분기
            dk.Dom().S( 'fontSmoothing', $v )
            dk.Dom().S( 'opacity', $v ) // ie8 이하 필터처리
dk.Dom 객체 삽입 및 제거 함수, html 태그 삽입함수, 문자열 삽입함수를 지원합니다. 
            '>' 큰쪽이 부모객체
            dk.Dom().S( '>', $dom )
            dk.Dom().S( '<', $dom )
            dk.Dom().S( '>-', $dom )
            dk.Dom().S( '<-', $dom )
            // innerHTML, 엘리먼트 앞 뒤 html 태그 삽입
            dk.Dom().S( 'html', $v )
            dk.Dom().S( '+html', $v )
            dk.Dom().S( 'html+', $v )
            // innerText, textContent 분기, 엘리먼트 앞 뒤 text 삽입
            dk.Dom().S( 'text', $v )
            dk.Dom().S( '+text', $v )
            dk.Dom().S( 'text+', $v )
이벤트 추가함수를 지원힙니다.
이벤트 키값, 핸들러 매서드
IE, 데스크탑, 모바일 자동 분기 - 해당 이벤트로 적용됩니다. 
            var cb = function( $e ){
                // $e.type, $e.target, $e.nativeEvent, $e.nativeTarget
                log( $e );
            }
            dk.Dom().S( 'over', cb ) // mouseover
            dk.Dom().S( 'enter', cb ) // mouseenter
            dk.Dom().S( 'out', cb ) // mouseout
            dk.Dom().S( 'leave', cb ) // mouseleave
            dk.Dom().S( 'click', cb )
            dk.Dom().S( 'dblclick', cb )
            dk.Dom().S( 'down', cb ) // mousedown, touchstart
            dk.Dom().S( 'move', cb ) // mousemove, touchmove
            dk.Dom().S( 'up', cb ) // mouseup, touchend
            dk.Dom().S( 'contextmenu', cb )

예제

dk.Css

스타일 객체 생성 ( CSSStyleSheet ) - dk.Css 객체

스타일시트의 룰객체를 이용하여 스타일시트 전반을 통제할 수 있습니다
css 생성 문법 그대로 적용됩니다. 
            dk.Css( 'html' ); // html{ }
            dk.Css( 'body' ); // body{ }
            dk.Css( 'html, body' ); // html, body{ }
            dk.Css( '.class' ); // .class{ }
            dk.Css( '#class' ); // #class{ }
            dk.Css( 'a:hover' ); // a:hover{ }

S method

dk.Dom의 S method와 동일한 인터페이스를 지원합니다.
key, value 시스템, css vendor prefix, 편의성 css 입력 함수 
			dk.Css( 'body' ).S( 'margin', 0, 'padding', 0 ); // 기본 css 문법
			dk.Css( 'body' ).S( 'color', '#808081', 'font', '12px/18px Helvetica, Arial' ); // 중복 설정 가능 (상속)
			dk.Css( 'body' ).S( 'bgColor', '#eceef4' ); // 편의성 css 입력 함수
			dk.Css( '.test' ).S( 'boxShadow', '0px 0px 10px #62615f inset' ); // css vendor prefix

			dk.Css( 'a:link, a:visited, a:active' ).S( 'color', '#808081', 'textDecoration', 'none' );
			dk.Css( 'a:hover' ).S( 'color', '#222', 'textDecoration', 'underline' );
			dk.Css( '.hr2' ).S( 'margin', '30px 0', 'borderTop', '1px solid #aaa' );

제어문 처리가능

dk.Css 객체를 통해 프로그래밍 언어로서 변수, 문, 식을 다양하게 응용할 수 있습니다. 
			// ie 일 경우 배경색 흰색, 아닐경우 배경색 검정
			var agent = window.navigator.userAgent.toLowerCase();
			if( agent.indexOf( 'msie' ) < 0 && agent.indexOf( 'trident' ) < 0 ){
				dk.Css( 'body' ).S( 'bgColor', '#fff' );
			} else {
				dk.Css( 'body' ).S( 'bgColor', '#000' );
			}

dk.DETECTOR

데스크탑, 모바일, 브라우저, 객체 탐지 등

Detector를 통해서 브라우저 및 객체탐지 등을 미리 파악할 수 있습니다. 
            dk.DETECTOR;
            
            
			// ie 일 경우 배경색 흰색, 아닐경우 배경색 검정
			if( dk.DECTOER.browser == 'ie' ){
				dk.Css( 'body' ).S( 'bgColor', '#fff' );
			} else {
				dk.Css( 'body' ).S( 'bgColor', '#000' );
			}
